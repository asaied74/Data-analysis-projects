# Delete the missing data
NaN_col = list()

def Check_NaN(column):
    NaN_sum = df[column].isnull().sum()
    if(NaN_sum > 700):
        NaN_col.append(column)
        print(f"Column '{column}' has {NaN_sum} Null values.")
    
for col in list(df.columns):
    Check_NaN(col)
df.drop(NaN_col, axis=1, inplace=True)
